3.3 — Commentaires

Concepts clés

  • Commentaire inutile : décrit CE QUE fait le code (visible sans commentaire)
  • Commentaire utile : explique POURQUOI le code fait ça, ou un algorithme non-évident
  • Auto-documentation : code si clair qu'il n'a pas besoin de commentaires
  • JSDoc : format standard pour documenter les fonctions : @param, @returns, @throws
  • TODO/FIXME : commentaires temporaires acceptables pendant le développement

Commentaire inutile vs utile — exemples

// INUTILE : décrit l'évidence
let age = 25; // assigne 25 à age
i++; // incrémente i
return false; // retourne false

// INUTILE : reformule le nom de la fonction
// Calcule la moyenne
function calculerMoyenne(notes) { ... }
// UTILE : explique une décision non-évidente
// On utilise Math.floor et non Math.round pour éviter les faux positifs
// lors de la vérification des droits d'accès par tranche d'âge
let tranche = Math.floor(age / 10) * 10;

// UTILE : algorithme complexe
// Algorithme de Luhn : valide un numéro de carte bancaire
// en alternant doubles et simples selon la position du chiffre
function validerCarte(numero) { ... }

function chiffrerCesar(texte, decalage) {
    return texte
        .split('')
        .map(char => {
            let code = char.charCodeAt(0);
            if (code >= 65 && code <= 90) {
                return String.fromCharCode(((code - 65 + decalage) % 26) + 65);
            }
            if (code >= 97 && code <= 122) {
                return String.fromCharCode(((code - 97 + decalage) % 26) + 97);
            }
            return char;
        })
        .join('');
}

function trouverPremiers(limite) {
    let candidats = Array.from({ length: limite - 1 }, (_, i) => i + 2);
    for (let i = 2; i <= Math.sqrt(limite); i++) {
        candidats = candidats.filter(n => n === i || n % i !== 0);
    }
    return candidats;
}

Bonus 2 — Transformer des commentaires en fonctions

Quand un commentaire explique ce QUE fait un bloc de code, souvent c'est signe qu'on devrait extraire ce bloc en fonction. Applique ça ici :

function traiterInscription(utilisateur) {
    // Vérifier que l'email est valide
    let emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
    if (!emailRegex.test(utilisateur.email)) {
        return { succes: false, erreur: 'Email invalide' };
    }

    // Vérifier que le mot de passe est assez fort
    if (utilisateur.motDePasse.length < 8 ||
        !/[A-Z]/.test(utilisateur.motDePasse) ||
        !/[0-9]/.test(utilisateur.motDePasse)) {
        return { succes: false, erreur: 'Mot de passe trop faible' };
    }

    // Générer un token de vérification
    let token = Math.random().toString(36).substring(2) +
                Date.now().toString(36);

    // Sauvegarder l'utilisateur
    utilisateur.actif = false;
    utilisateur.tokenVerification = token;
    utilisateur.dateInscription = new Date();

    return { succes: true, token };
}

Bonus 3 — Documenter une recette algorithmique

Cet algorithme de recherche binaire est un classique. Documente-le avec des commentaires de haut niveau qui expliquent la stratégie, pas les lignes individuelles.

Piste : explique la stratégie "diviser pour régner", pourquoi on calcule mid ainsi, et les 3 cas possibles.

function rechercheBinaire(tableau, cible) {
    let gauche = 0;
    let droite = tableau.length - 1;

    while (gauche <= droite) {
        let milieu = Math.floor((gauche + droite) / 2);

        if (tableau[milieu] === cible) {
            return milieu;
        }

        if (tableau[milieu] < cible) {
            gauche = milieu + 1;
        } else {
            droite = milieu - 1;
        }
    }

    return -1;
}

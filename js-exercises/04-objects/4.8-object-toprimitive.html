4.8 — Conversion d'objet en primitif

Concepts clés

  • Hints : quand JS a besoin de convertir un objet en primitif, il choisit un "hint" :
    • "string" : contexte string (template literal, alert, concaténation explicite)
    • "number" : contexte numérique (+obj unaire, comparaisons, opérations math)
    • "default" : contexte ambigu (obj + something, ==)
  • Symbol.toPrimitive(hint) : méthode prioritaire pour contrôler toutes les conversions
  • toString() : appelé pour hint "string" (ou en fallback)
  • valueOf() : appelé pour hint "number" et "default" (ou en fallback)
  • Ordre de priorité : Symbol.toPrimitive → valueOf/toString selon le hint → erreur
// Sans Symbol.toPrimitive — utilise toString/valueOf
let obj = {
    value: 42,
    toString() { return "objet(42)"; },
    valueOf() { return 42; }
};

alert(obj);      // "objet(42)" — hint string → toString()
console.log(+obj);    // 42 — hint number → valueOf()
console.log(obj + 1); // 43 — hint default → valueOf()

// Avec Symbol.toPrimitive — contrôle total
let obj2 = {
    [Symbol.toPrimitive](hint) {
        if (hint === "string") return "je suis un objet";
        return 100; // number et default
    }
};

B2 — Objet Fraction avec arithmétique

Crée un objet Fraction avec des méthodes add(other), multiply(other), et les conversions : toString() retourne "a/b" et valueOf() retourne la valeur décimale. Simplifie automatiquement avec le PGCD. 

// Ta réponse ici

// Calcul du PGCD (Plus Grand Commun Diviseur)
function gcd(a, b) {
    return b === 0 ? a : gcd(b, a % b);
}

function createFraction(numerator, denominator) {
    if (denominator === 0) throw new Error("Dénominateur ne peut pas être 0");

    // Simplification automatique
    let divisor = gcd(Math.abs(numerator), Math.abs(denominator));
    let n = numerator / divisor;
    let d = denominator / divisor;

    return {
        n,
        d,

        add(other) {
            // a/b + c/d = (a*d + c*b) / (b*d)

        },

        multiply(other) {
            // a/b * c/d = (a*c) / (b*d)

        },

        toString() {
            return `${this.n}/${this.d}`;
        },

        valueOf() {
            return this.n / this.d;
        }
    };
}

let half = createFraction(1, 2);
let third = createFraction(1, 3);

console.log(`${half}`);                        // "1/2"
console.log(+half);                             // 0.5
console.log(`${half.add(third)}`);             // "5/6"
console.log(`${half.multiply(third)}`);        // "1/6"
console.log(createFraction(6, 4).toString());  // "3/2" (simplifié)
console.log(half > third);                      // true (0.5 > 0.333...)

Le PGCD (algorithme d'Euclide) est un fondamental en algorithmique. Mérite une note Obsidian !

B3 — Objet Duration avec double conversion

Crée un objet Duration qui représente une durée en secondes. En mode "number" retourne les secondes totales, en mode "string" retourne "Xh Ym Zs" en omettant les zéros (ex: "1h 30m", "45s", "2h 5s"). 

// Ta réponse ici
function createDuration(totalSeconds) {
    return {
        totalSeconds,

        [Symbol.toPrimitive](hint) {
            if (hint === "string") {
                let h = Math.floor(this.totalSeconds / 3600);
                let m = Math.floor((this.totalSeconds % 3600) / 60);
                let s = this.totalSeconds % 60;

                // Construire la string en omettant les parties à 0
                let parts = [];
                // if (h > 0) parts.push(...)
                // if (m > 0) parts.push(...)
                // if (s > 0) parts.push(...)

                return parts.join(" ") || "0s";

            } else {
                return this.totalSeconds;
            }
        },

        add(other) {
            return createDuration(this.totalSeconds + +other);
        }
    };
}

let d1 = createDuration(3600 + 1800);    // 1h 30m
let d2 = createDuration(45);              // 45s
let d3 = createDuration(7200 + 5);        // 2h 5s
let d4 = createDuration(0);               // 0s

console.log(`${d1}`);    // "1h 30m"
console.log(`${d2}`);    // "45s"
console.log(`${d3}`);    // "2h 5s"
console.log(`${d4}`);    // "0s"

console.log(+d1);         // 5400
console.log(d1 > d2);     // true

let total = d1.add(d2);
console.log(`${total}`);  // "1h 30m 45s"

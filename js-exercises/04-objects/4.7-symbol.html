4.7 — Type Symbol

Concepts clés

  • Symbol() : crée un identifiant unique et immuable. Deux symboles avec la même description ne sont jamais égaux.
  • Description : Symbol("id") — la description est juste un label pour le debug, pas l'identifiant
  • Clés d'objet : les symboles peuvent être des clés, accessibles avec bracket notation : obj[sym]
  • Invisibles aux itérations : for..in, Object.keys() et JSON.stringify() ignorent les propriétés symboliques
  • Symbol.for(key) : registre global — retourne toujours le même symbole pour la même clé
  • Symboles système : Symbol.iterator, Symbol.toPrimitive, etc. — permettent de customiser le comportement des objets
// Chaque Symbol() est unique
let id1 = Symbol("id");
let id2 = Symbol("id");
console.log(id1 === id2); // false — même description, mais symboles différents

// Utilisation comme clé cachée
let user = { name: "Alice" };
let id = Symbol("id");
user[id] = 123;           // clé symbolique invisible aux itérations

console.log(user[id]);    // 123
for (let key in user) {
    console.log(key);     // affiche seulement "name", jamais le Symbol
}

// Symbol.for() — registre global partagé
let s1 = Symbol.for("shared");
let s2 = Symbol.for("shared");
console.log(s1 === s2); // true — même symbole global

Exercice 3 — Symbol.for() pour les symboles partagés

Crée deux "modules" simulés qui doivent partager le même symbole pour communiquer sans collision de noms. Utilise Symbol.for(). 

// Module A — crée ou accède au symbole global
const moduleA = (function() {
    const STATUS = Symbol.for("app.status"); // registre global

    return {
        setStatus(obj, status) {
            obj[STATUS] = status;
        }
    };
})();

// Module B — accède au MÊME symbole global sans importer depuis A
const moduleB = (function() {
    const STATUS = Symbol.for("app.status"); // même clé → même symbole

    return {
        getStatus(obj) {
            return obj[STATUS];
        }
    };
})();

// Test : les deux modules partagent le même symbole
let app = {};
moduleA.setStatus(app, "running");
console.log(moduleB.getStatus(app)); // "running"

// Vérification : c'est bien le même symbole
let symA = Symbol.for("app.status");
let symB = Symbol.for("app.status");
console.log(symA === symB); // true

// Symbol.keyFor() pour retrouver la clé d'un symbole global
console.log(Symbol.keyFor(symA)); // "app.status"

// Contrairement aux symboles locaux :
let localSym = Symbol("local");
console.log(Symbol.keyFor(localSym)); // undefined — pas dans le registre global

B3 — Propriétés "privées" avec symboles

Crée un module UserManager qui stocke des données sensibles (mot de passe hashé, token) dans des propriétés symboliques, invisibles pour le code externe. 

// Ta réponse ici

const UserManager = (function() {
    // Ces symboles ne sont accessibles que dans ce module
    const PASSWORD_HASH = Symbol("passwordHash");
    const TOKEN = Symbol("token");

    function hashPassword(password) {
        // Simulation d'un hash (en vrai, utiliser bcrypt)
        return btoa(password + "_hashed"); // btoa = base64 encode
    }

    return {
        createUser(name, password) {
            let user = {
                name,
                createdAt: new Date().toISOString()
            };
            // Stocker le hash dans une propriété symbolique
            user[PASSWORD_HASH] = hashPassword(password);
            return user;
        },

        authenticate(user, password) {
            // Vérifier le mot de passe
            return user[PASSWORD_HASH] === hashPassword(password);
        },

        setToken(user, token) {
            user[TOKEN] = token;
        },

        getToken(user) {
            return user[TOKEN];
        }
    };
})();

// Tests
let user = UserManager.createUser("Alice", "motdepasse123");

console.log(user.name);       // "Alice" — visible
console.log(JSON.stringify(user)); // pas de passwordHash ni token — sécurisé !

console.log(UserManager.authenticate(user, "motdepasse123")); // true
console.log(UserManager.authenticate(user, "mauvais"));        // false

UserManager.setToken(user, "jwt_token_xyz");
console.log(UserManager.getToken(user)); // "jwt_token_xyz"

// Code externe ne peut pas accéder aux propriétés symboliques
// (sans avoir la référence au Symbol)
console.log(Object.keys(user)); // ["name", "createdAt"] — sécurisé

Ce pattern est utile pour séparer les données publiques des données internes. En JavaScript moderne, les vraies propriétés privées s'écrivent avec # dans les classes (ES2022).

5.10 — Affectation par décomposition

Décomposition de tableaux

let [a, b, c] = [1, 2, 3]; // a=1, b=2, c=3

// Ignorer des éléments avec des virgules
let [first, , third] = [1, 2, 3]; // first=1, third=3

// Valeurs par défaut
let [x = 10, y = 20] = [5]; // x=5, y=20

// Rest operator
let [head, ...tail] = [1, 2, 3, 4]; // head=1, tail=[2,3,4]

// Échange de variables (élégant !)
let m = 1, n = 2;
[m, n] = [n, m]; // m=2, n=1

Décomposition d'objets

let { name, age } = { name: "Jean", age: 30, ville: "Paris" };
// name="Jean", age=30 — ville est ignorée

// Renommage
let { name: userName, age: userAge } = user;
// userName="Jean", userAge=30

// Valeurs par défaut
let { name = "Anonyme", role = "user" } = { name: "Jean" };
// name="Jean", role="user" (valeur par défaut)

// Rest
let { name, ...rest } = { name: "Jean", age: 30, ville: "Paris" };
// name="Jean", rest={ age:30, ville:"Paris" }

Dans les paramètres de fonctions

// Décomposer directement dans la signature de la fonction
function greet({ name, age = 0 }) {
    console.log(`${name} a ${age} ans`);
}
greet({ name: "Jean", age: 30 });

Fondamental : la décomposition est syntaxique — elle ne modifie pas l'objet original, elle crée juste des variables locales.
